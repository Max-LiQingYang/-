<script type="text/babel">

      // 封装组件

      class App extends React.Component {

        constructor () {

          super();

          this.state = {

            message: "hello world"

          };

        }

  

        // render 永远不要手动调用

        render() {

          return (

            <div>

              <h2>{this.state.message}</h2>

              // 对于 React 的组件类

              // 在这里的 this 绑定的是 undefined 而不是类,

              // btnClick.apply(undefined)

              // 所以这里需要改变 this 的指向, 指向这个类, 才能使用这个类的方法

              <button onClick={this.btnClick.bind(this)}>改变文本</button>

            </div>

          );

        }

  

        btnClick () {

          // this.message = "xixi";

          // this.state.message = "hello";

          // render 函数不会自动重新执行

  

          // 响应式改变

          this.setState({

            message: "Hello React"

          });

        }

      }

  

      ReactDOM.render(<App />, document.getElementById("app"));

    </script>
